This article is gonna show you how to get stock real-time information through python

import urllib2
import urllib
import re
import xml.etree.cElementTree as ET
from lxml import etree 
from win32com import client as ax
import time



debug=False
stockList = []
i = 0

def ToGB(str):
    if(debug): print(str)
    return str.decode('gb2312')

def getStockInfo(url):
    try:

        request = urllib2.Request(url)
        reponse = urllib2.urlopen(request)
        return reponse.read()
    except Exception:
        raise "no such url"



def ParseResultStr(resultstr):
    if(debug) : print(resultstr)
    slist=resultstr.split(',')
    m=re.match(r'(.+)=',slist[0])
    header = m.group()
    stocknumber = header[-9:-1]
    stockchart = stocknumber+".gif"
    print stockchart
    name = slist[0][len(header)+1:len(slist[0])]

    yesterdayendprice=slist[2]
    todaystartprice=slist[1]
    nowprice=slist[3]
    upgraderate=(float(nowprice)-float(yesterdayendprice))/float(yesterdayendprice)
    upgraderate= str("%.2f" % (upgraderate * 100))
    #print upgraderate
    dateandtime=slist[30] + ' ' + slist[31]
    imgurl = "http://image.sinajs.cn/newchart/daily/n/"+stockchart
    urllib.urlretrieve(imgurl,r"C:\Users\Public\Documents\Scala\LocalIntegratedContent"+"\\"+stockchart)
    return(stockchart,stocknumber,name,yesterdayendprice,todaystartprice,nowprice,upgraderate,dateandtime)

def getStockList(num):

    for stockNum in range(600000+num,600000+num+79):
       url = "http://hq.sinajs.cn/list="+"sh"+str(stockNum)
       info = getStockInfo(url)
       m = re.match(r'(.+)=',info)
       header = m.group()
       stockName = info[len(header):len(header)+2]
       if stockName!='""':
          stockList.append(ParseResultStr(ToGB(info)))
    return stockList


def createXML(datalist):
    root = ET.Element("stockdata")
    for datainfo in datalist:
        stock = ET.SubElement(root,"stock")
        ET.SubElement(stock,"stockchart").text,ET.SubElement(stock,"stocknumber").text,ET.SubElement(stock,"name").text,ET.SubElement(stock,"yesterdayendprice").text,ET.SubElement(stock,"todaystartprice").text,ET.SubElement(stock,"nowprice").text,ET.SubElement(stock,"upgraderate").text,ET.SubElement(stock,"dateandtime").text=datainfo
    tree = ET.ElementTree(root)
    tree.write("stock.xml")
    return "stock.xml"


def prettyXML(xml):
    parser = etree.XMLParser(
        remove_blank_text=False, resolve_entities=True, strip_cdata=True)
    xmlfile = etree.parse(xml, parser)
    pretty_xml = etree.tostring(
        xmlfile, encoding = 'UTF-8', xml_declaration = True, pretty_print = True
        )
    file = open(xml, "w")
    file.writelines(pretty_xml)
    file.close()


try:
    obj = ax.Dispatch("Scala.InfoNetwork")
except Exception,e:
    raise "%s Error:could not create Scala.InfoNetwork object"%e

while True:
    prettyXML(createXML(getStockList(i)))

    obj.IntegrateContentLocally(r"C:\Users\Administrator\Desktop\stock.xml","")
    time.sleep(15)

    if i<=1921:
        i = i+79
    else:
        i = 0
